STM32関数集

関数一覧表

よく使う関数をまとめました

コピペように使ってね

GPIO

PinLevel指定

HAL_GPIO_WritePin(GPIOx,GPIO_PIN_y, GPIO_PIN_SET);

PinLevel反転

HAL_GPIO_TogglePin(GPIOx,GPIO_PIN_y);

UART

通常送受信

HAL_UART_Transmit(&huartx, pdata, len, timeout);
HAL_UART_Receive(&huartx, pdata, len, timeout);

割り込み送受信

HAL_UART_Transmit_IT(&huartx, pdata, len);
HAL_UART_Receive_IT(&huartx, pdata, len);

DMA割り込み送受信

HAL_UART_Transmit_DMA(&huartx, pdata, len);
HAL_UART_Receive_DMA(&huartx, pdata, len);

コールバック関数

割り込み終了時に呼び出される関数

引数に送受信完了したhuartのポインタが入っているので判定が可能

void HAL_UART_TxCpltCallback(UART_HandleTypeDef *huart){
            
    if(huart = &huartx){

        // 送信完了時の処理
    }
}
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart){
            
    if(huart = &huartx){

        // 受信完了時の処理
    }
}
